```typescript
// src/services/api.ts
import axios, { type AxiosInstance, type AxiosRequestConfig } from 'axios'
import { useUserStore } from '@/stores/user'

class ApiClient {
  private client: AxiosInstance
  
  constructor(baseURL: string) {
    this.client = axios.create({
      baseURL,
      timeout: 30000,
      headers: {
        'Content-Type': 'application/json'
      }
    })
    
    this.setupInterceptors()
  }
  
  private setupInterceptors() {
    // 请求拦截器
    this.client.interceptors.request.use(
      (config) => {
        const userStore = useUserStore()
        if (userStore.token) {
          config.headers.Authorization = `Bearer ${userStore.token}`
        }
        return config
      },
      (error) => Promise.reject(error)
    )
    
    // 响应拦截器
    this.client.interceptors.response.use(
      (response) => response.data,
      (error) => {
        if (error.response?.status === 401) {
          // 处理未授权
          const userStore = useUserStore()
          userStore.logout()
        }
        return Promise.reject(error)
      }
    )
  }
  
  async get<T>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.client.get(url, config)
  }
  
  async post<T>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.client.post(url, data, config)
  }
  
  async put<T>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.client.put(url, data, config)
  }
  
  async delete<T>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.client.delete(url, config)
  }
}

export const api = new ApiClient(import.meta.env.VITE_API_BASE_URL)
```

### 5.2 Service层实现

```typescript
// src/services/chat.service.ts
import { api } from './api'
import type { Message, ChatSession, ChatRequest } from '@/types/chat.types'

export const chatService = {
  // 获取会话列表
  async getSessions(): Promise<ChatSession[]> {
    return api.get('/api/v1/chat/sessions')
  },
  
  // 创建新会话
  async createSession(title?: string): Promise<ChatSession> {
    return api.post('/api/v1/chat/sessions', { title })
  },
  
  // 发送消息
  async sendMessage(request: ChatRequest): Promise<Message> {
    return api.post('/api/v1/chat', request)
  },
  
  // 流式发送消息
  async sendMessageStream(
    request: ChatRequest,
    onChunk: (chunk: string) => void,
    onDone: () => void,
    onError: (error: Error) => void
  ): Promise<AbortController> {
    const controller = new AbortController()
    
    const response = await fetch(`${import.meta.env.VITE_API_BASE_URL}/api/v1/chat/stream`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${useUserStore().token}`
      },
      body: JSON.stringify(request),
      signal: controller.signal
    })
    
    const reader = response.body?.getReader()
    const decoder = new TextDecoder()
    
    if (reader) {
      const pump = async () => {
        try {
          const { done, value } = await reader.read()
          if (done) {
            onDone()
            return
          }
          
          const chunk = decoder.decode(value)
          onChunk(chunk)
          await pump()
        } catch (error) {
          onError(error as Error)
        }
      }
      
      pump()
    }
    
    return controller
  },
  
  // 获取会话消息
  async getSessionMessages(sessionId: string): Promise<Message[]> {
    return api.get(`/api/v1/chat/sessions/${sessionId}/messages`)
  },
  
  // 删除会话
  async deleteSession(sessionId: string): Promise<void> {
    return api.delete(`/api/v1/chat/sessions/${sessionId}`)
  }
}
```

## 6. 状态管理规范

### 6.1 Pinia Store 定义

```typescript
// src/stores/chat.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { chatService } from '@/services/chat.service'
import type { Message, ChatSession } from '@/types/chat.types'

export const useChatStore = defineStore('chat', () => {
  // 状态
  const sessions = ref<ChatSession[]>([])
  const currentSessionId = ref<string | null>(null)
  const messages = ref<Message[]>([])
  const isLoading = ref(false)
  const error = ref<string | null>(null)
  
  // 计算属性
  const currentSession = computed(() => 
    sessions.value.find(s => s.id === currentSessionId.value)
  )
  
  const hasMessages = computed(() => messages.value.length > 0)
  
  // 方法
  async function initializeSession(sessionId?: string) {
    if (sessionId) {
      currentSessionId.value = sessionId
      await loadSession(sessionId)
    } else if (sessions.value.length === 0) {
      await createNewSession()
    }
  }
  
  async function loadSession(sessionId: string) {
    try {
      isLoading.value = true
      error.value = null
      
      const sessionMessages = await chatService.getSessionMessages(sessionId)
      messages.value = sessionMessages
      currentSessionId.value = sessionId
    } catch (err) {
      error.value = '加载会话失败'
      console.error(err)
    } finally {
      isLoading.value = false
    }
  }
  
  async function createNewSession(title?: string) {
    try {
      const session = await chatService.createSession(title)
      sessions.value.unshift(session)
      currentSessionId.value = session.id
      messages.value = []
    } catch (err) {
      error.value = '创建会话失败'
      throw err
    }
  }
  
  async function sendMessage(content: string) {
    if (!currentSessionId.value || isLoading.value) return
    
    const userMessage: Message = {
      id: crypto.randomUUID(),
      sessionId: currentSessionId.value,
      role: 'user',
      content,
      timestamp: new Date().toISOString()
    }
    
    messages.value.push(userMessage)
    isLoading.value = true
    
    try {
      const aiMessage = await chatService.sendMessage({
        sessionId: currentSessionId.value,
        content,
        context: getContext()
      })
      
      messages.value.push(aiMessage)
    } catch (err) {
      error.value = '发送消息失败'
      // 移除失败的消息
      messages.value.pop()
    } finally {
      isLoading.value = false
    }
  }
  
  function getContext() {
    // 获取当前代码上下文等
    return {}
  }
  
  function clearError() {
    error.value = null
  }
  
  return {
    // 状态
    sessions,
    currentSessionId,
    messages,
    isLoading,
    error,
    
    // 计算属性
    currentSession,
    hasMessages,
    
    // 方法
    initializeSession,
    loadSession,
    createNewSession,
    sendMessage,
    clearError
  }
}, {
  persist: {
    key: 'heliora-chat',
    paths: ['sessions', 'currentSessionId']
  }
})
```
