## 0. 适用范围与独立化约束

1. 本文档是后端领域规范，通用工程规则以《00-工程与代码规范总则》为准。
2. 后端核心业务（编排、检索、教学策略、文件安全）必须自研，不直接复制第三方业务实现。
3. 工具执行层采用可插拔设计，允许兼容外部执行方案，但不对单一外部项目形成硬依赖。
4. 新增第三方依赖前必须完成许可证审查，并在依赖清单登记许可证与引入理由。

## 5. 服务层设计

### 5.1 引导引擎实现

```python
# app/services/chat/guide_engine.py
from enum import Enum
from typing import Optional, List
from pydantic import BaseModel

class GuideStrategy(Enum):
    CLARIFY = "clarify"
    LOCATE = "locate"
    REFLECT = "reflect"
    SCAFFOLD = "scaffold"
    FADE = "fade"

class GuideState(BaseModel):
    current_strategy: GuideStrategy
    hint_level: int = 1  # 0-3
    attempt_count: int = 0
    last_question: Optional[str] = None

class GuideEngine:
    """引导式教学引擎."""
    
    def __init__(self, llm_client, graph_service):
        self.llm = llm_client
        self.graph = graph_service
        
    async def generate_question(
        self,
        error_context: dict,
        student_answer: Optional[str] = None,
        state: Optional[GuideState] = None
    ) -> tuple[str, GuideState]:
        """生成引导式问题."""
        
        if state is None:
            state = GuideState(current_strategy=GuideStrategy.CLARIFY)
        
        # 检索相关知识
        knowledge = await self.graph.search_related_concepts(
            error_type=error_context.get('error_type'),
            code_context=error_context.get('code')
        )
        
        # 根据状态选择策略
        if student_answer is None:
            # 首次提问
            question = await self._generate_initial_question(
                error_context, knowledge, state.current_strategy
            )
        else:
            # 评估回答
            evaluation = await self._evaluate_answer(student_answer, error_context)
            
            if evaluation.is_correct:
                # 回答正确，进入下一阶段
                state = self._advance_strategy(state)
                question = await self._generate_follow_up_question(
                    error_context, knowledge, state.current_strategy, evaluation
                )
            else:
                # 回答错误，增加提示或降级策略
                state.attempt_count += 1
                
                if state.attempt_count >= 2:
                    # 连续失败，降级提示
                    state.hint_level = min(state.hint_level + 1, 3)
                
                question = await self._generate_hint_question(
                    error_context, knowledge, student_answer, state.hint_level
                )
        
        state.last_question = question
        return question, state
    
    async def _generate_initial_question(
        self,
        error_context: dict,
        knowledge: list,
        strategy: GuideStrategy
    ) -> str:
        """生成初始问题."""
        
        prompts = {
            GuideStrategy.CLARIFY: self._clarify_prompt,
            GuideStrategy.LOCATE: self._locate_prompt,
            GuideStrategy.REFLECT: self._reflect_prompt,
        }
        
        prompt_template = prompts.get(strategy, self._clarify_prompt)
        
        response = await self.llm.generate(
            prompt=prompt_template(error_context, knowledge),
            temperature=0.7,
            max_tokens=200
        )
        
        return response.content
    
    def _advance_strategy(self, state: GuideState) -> GuideState:
        """推进到下一阶段."""
        strategy_order = [
            GuideStrategy.CLARIFY,
            GuideStrategy.LOCATE,
            GuideStrategy.REFLECT,
            GuideStrategy.SCAFFOLD,
            GuideStrategy.FADE
        ]
        
        current_index = strategy_order.index(state.current_strategy)
        if current_index < len(strategy_order) - 1:
            state.current_strategy = strategy_order[current_index + 1]
        state.attempt_count = 0
        
        return state
```

### 5.2图谱构建服务

```python
# app/services/graph/builder.py
from typing import List, Optional
import asyncio
from tree_sitter import Language, Parser

class GraphBuilder:
    """知识图谱构建服务."""
    
    def __init__(self, neo4j_driver, llm_client):
        self.driver = neo4j_driver
        self.llm = llm_client
        self.parsers = self._initialize_parsers()
    
    async def build_from_course_material(
        self,
        material_path: str,
        course_code: str
    ) -> dict:
        """从课程材料构建图谱."""
        
        # 1. 解析文档
        sections = await self._parse_document(material_path)
        
        # 2. 提取概念
        concepts = await self._extract_concepts(sections)
        
        # 3. 建立关系
        relationships = await self._build_relationships(concepts)
        
        # 4. 写入图谱
        result = await self._write_to_graph(concepts, relationships, course_code)
        
        return result
    
    async def build_from_code(
        self,
        code_path: str,
        language: str,
        project_id: str
    ) -> dict:
        """从代码构建图谱."""
        
        # 1. 解析代码 AST
        ast_nodes = await self._parse_code(code_path, language)
        
        # 2. 提取CodeUnit
        code_units = self._extract_code_units(ast_nodes, code_path, project_id)
        
        # 3. 分析依赖关系
        dependencies = await self._analyze_dependencies(code_units)
        
        # 4. 计算复杂度指标
        metrics = await self._calculate_metrics(code_units)
        
        # 5. 写入图谱
        result = await self._write_code_units(code_units, dependencies, metrics)
        
        return result
    
    async def _parse_code(self, file_path: str, language: str) -> List[dict]:
        """解析代码文件."""
        parser = self.parsers.get(language)
        if not parser:
            raise ValueError(f"Unsupported language: {language}")
        
        with open(file_path, 'rb') as f:
            source_code = f.read()
        
        tree = parser.parse(source_code)
        return self._traverse_ast(tree.root_node, source_code)
    
    def _extract_code_units(
        self,
        ast_nodes: List[dict],
        file_path: str,
        project_id: str
    ) -> List[CodeUnit]:
        """提取代码单元."""
        code_units = []
        
        for node in ast_nodes:
            if node['type'] in ['class_definition', 'function_definition']:
                code_unit = CodeUnit(
                    id=f"code_{project_id}_{node['name']}",
                    name=node['name'],
                    type=node['type'],
                    file_path=file_path,
                    start_line=node['start_point'][0],
                    end_line=node['end_point'][0],
                    signature=node.get('signature', ''),
                    language=self._detect_language(file_path)
                )
                code_units.append(code_unit)
        
        return code_units
```

class GraphQueries:
    """图谱查询封装."""
    
    FIND_CONCEPT_BY_ID = """
    MATCH (c:Concept {id: $concept_id})
    RETURN c
    """
    
    FIND_RELATED_CONCEPTS = """
    MATCH (c:Concept {id: $concept_id})-[:PREREQUISITE_OF|RELATED_TO*1..2]-(related:Concept)
    RETURN related, 
           relationships(related) as relationships,
           length(shortestPath((c)-[*]-(related))) as distance
    ORDER BY distance ASC
    LIMIT $limit
    """
    
    FIND_CODE_BY_CONCEPT = """
    MATCH (c:Concept {id: $concept_id})-[:EXPLAINS]->(code:CodeUnit)
    RETURN code
    ORDER BY code.confidence DESC
    """
    
    FIND_BUG_HISTORY = """
    MATCH (bug:BugCase)-[:FIXED_BY]->(code:CodeUnit {file_path: $file_path})
    RETURN bug
    ORDER BY bug.occurred_at DESC
    LIMIT $limit
    """
    
    GET_LEARNING_PATH = """
    MATCH path = (target:Concept {id: $concept_id})-[:PREREQUISITE_OF*]->(prereq:Concept)
    WHERE NOT (prereq)-[:PREREQUISITE_OF]->()
    WITH nodes(path) as concepts, length(path) as depth
    ORDER BY depth DESC
    UNWIND concepts as concept
    RETURN concept, concept.mastery_level as mastery
    """
    
    CREATE_CONCEPT = """
    CREATE (c:Concept $properties)
    RETURN c
    """
    
    CREATE_RELATIONSHIP = """
    MATCH (a {id: $from_id}), (b {id: $to_id})
    MERGE (a)-[r:$rel_type]->(b)
    SET r += $properties
    RETURN r
    """
