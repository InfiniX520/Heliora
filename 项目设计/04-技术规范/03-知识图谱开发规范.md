```python
# app/db/neo4j/driver.py
from neo4j import GraphDatabase, AsyncGraphDatabase
from typing import Optional
import asyncio

class Neo4jManager:
    """Neo4j 驱动管理器."""
    
    _instance: Optional['Neo4jManager'] = None
    
    def __init__(self, uri: str, user: str, password: str):
        self.uri = uri
        self.user = user
        self.password = password
        self._driver = None
        self._async_driver = None
    
    @classmethod
    def get_instance(cls, uri: str, user: str, password: str) -> 'Neo4jManager':
        """获取单例实例."""
        if cls._instance is None:
            cls._instance = cls(uri, user, password)
        return cls._instance
    
    def get_driver(self):
        """获取同步驱动."""
        if self._driver is None:
            self._driver = GraphDatabase.driver(
                self.uri,
                auth=(self.user, self.password),
                max_connection_pool_size=50,
                connection_acquisition_timeout=60
            )
        return self._driver
    
    async def get_async_driver(self):
        """获取异步驱动."""
        if self._async_driver is None:
            self._async_driver = AsyncGraphDatabase.driver(
                self.uri,
                auth=(self.user, self.password),
                max_connection_pool_size=50
            )
        return self._async_driver
    
    async def close(self):
        """关闭驱动."""
        if self._driver:
            self._driver.close()
        if self._async_driver:
            await self._async_driver.close()

# 依赖注入
async def get_neo4j_session():
    """获取Neo4j 会话."""
    manager = Neo4jManager.get_instance(
        uri=settings.NEO4J_URI,
        user=settings.NEO4J_USER,
        password=settings.NEO4J_PASSWORD
    )
    driver = await manager.get_async_driver()
    async with driver.session() as session:
        yield session
```

## 2. Cypher查询规范

### 2.1查询编写规范

**命名规范**:
```cypher
//  正确：节点变量使用有意义的名称
MATCH (concept:Concept {id: $concept_id})
MATCH (code:CodeUnit)-[:IMPLEMENTS]->(concept)
RETURN code, concept

//  错误：无意义的单字母
MATCH (c:Concept {id: $id})
MATCH (x:CodeUnit)-[:IMPLEMENTS]->(c)
RETURN x, c
```

**参数化查询**:
```cypher
//  正确：使用参数
MATCH (c:Concept)-[:PREREQUISITE_OF*1..$depth]-(target:Concept {id: $target_id})
WHERE c.mastery_level >= $min_mastery
RETURN c, target

//  错误：字符串拼接（SQL注入风险）
MATCH (c:Concept {id: '" + concept_id + "'})
RETURN c
```

**索引使用**:
```cypher
// 创建索引
CREATE INDEX concept_id_index IF NOT EXISTS FOR (c:Concept) ON (c.id)
CREATE INDEX code_file_path_index IF NOT EXISTS FOR (code:CodeUnit) ON (code.file_path)
CREATE INDEX bug_error_type_index IF NOT EXISTS FOR (bug:BugCase) ON (bug.error_type)

// 创建全文索引（用于搜索）
CREATE FULLTEXT INDEX concept_name_index FOR (c:Concept) ON EACH [c.name, c.description]
```

### 2.2常用查询模式

**模式 1: 查找前置知识链**
```cypher
// 查找某个概念的所有前置知识（递归）
MATCH path = (target:Concept {id: $concept_id})-[:PREREQUISITE_OF*]->(prereq:Concept)
WHERE NOT (prereq)-[:PREREQUISITE_OF]->()
WITH path, length(path) as depth
ORDER BY depth DESC
UNWIND nodes(path) as concept
RETURN DISTINCT concept, 
       concept.mastery_level as mastery,
       concept.difficulty as difficulty
ORDER BY concept.difficulty ASC
```

**模式 2: 查找相关代码示例**
```cypher
// 查找与概念相关的代码示例，按置信度排序
MATCH (concept:Concept {id: $concept_id})-[:EXPLAINS]->(code:CodeUnit)
WHERE code.confidence >= $dynamic_confidence_tau
OPTIONAL MATCH (code)-[:HAS_BUG]->(bug:BugCase)
WITH concept, code, count(bug) as bug_count
RETURN code, 
       code.confidence as confidence,
       bug_count,
       code.complexity_cyclomatic as complexity
ORDER BY code.confidence DESC, bug_count ASC
LIMIT 10
```

**模式 3: 查找相似错误案例**
```cypher
// 查找相似的错误案例
MATCH (current_bug:BugCase {id: $bug_id})
MATCH (similar:BugCase)
WHERE similar.error_type = current_bug.error_type
  AND similar.id != current_bug.id
WITH current_bug, similar,
     gds.similarity.cosine(
       current_bug.context_vector,
       similar.context_vector
     ) as similarity
WHERE similarity > 0.7
RETURN similar, similarity
ORDER BY similarity DESC
LIMIT 5
```

**模式4: 构建学习路径**
```cypher
// 构建从当前水平到目标概念的学习路径
MATCH (current:Concept {mastery_level: $current_level})
MATCH (target:Concept {id: $target_id})
MATCH path = shortestPath((current)-[:PREREQUISITE_OF*]-(target))
WITH path, nodes(path) as concepts
UNWIND concepts as concept
RETURN concept, 
       concept.difficulty as difficulty,
       concept.estimated_study_time as study_time
ORDER BY concept.difficulty ASC
```

**模式5: 记忆图扩展召回(P1标准组件)**
```cypher
// 由记忆候选节点扩展到多跳证据链，供 MemoryRetriever 重排
MATCH (m:Memory {id: $memory_id})-[:MENTIONS]->(seed:Concept)
CALL {
    WITH seed
    MATCH p=(seed)-[:RELATED_TO|PREREQUISITE_OF*1..2]-(near:Concept)
    RETURN near, min(length(p)) as hop
}
OPTIONAL MATCH (near)-[:EXPLAINS]->(code:CodeUnit)
RETURN near.id as concept_id,
             hop,
             collect(code.id)[0..5] as support_code_ids
ORDER BY hop ASC
LIMIT 20
```

## 3. 图谱构建流程

### 3.1从课程材料构建

```python
# app/services/graph/builders/course_builder.py
from typing import List, Dict
import asyncio
from app.db.neo4j.driver import get_neo4j_session
from app.parsers.document_parser import DocumentParser

class CourseGraphBuilder:
    """课程知识图谱构建器."""
    
    def __init__(self, document_parser: DocumentParser):
        self.parser = document_parser
    
    async def build_from_pdf(self, pdf_path: str, course_code: str) -> Dict:
        """从PDF 教材构建图谱."""
        
        # 1. 解析 PDF
        sections = await self.parser.parse_pdf(pdf_path)
        
        # 2. 提取概念
        concepts = await self._extract_concepts(sections)
        
        # 3. 建立关系
        relationships = await self._build_concept_relationships(concepts)
        
        # 4. 写入图谱
        async for session in get_neo4j_session():
            # 创建节点
            await self._create_concepts(session, concepts, course_code)
            
            # 创建关系
            await self._create_relationships(session, relationships)
        
        return {
            'concepts_created': len(concepts),
            'relationships_created': len(relationships)
        }
    
    async def _extract_concepts(self, sections: List[Dict]) -> List[Dict]:
        """从章节提取概念."""
        concepts = []
        
        for section in sections:
            # 使用 LLM 提取概念
            extracted = await self.llm.extract_concepts(
                content=section['content'],
                title=section['title']
            )
            
            for concept_data in extracted:
                concept = {
                    'id': f"concept_{course_code}_{concept_data['name'].lower().replace(' ', '-')}",
                    'name': concept_data['name'],
                    'description': concept_data.get('description', ''),
                    'difficulty': concept_data.get('difficulty', 1),
                    'course_code': course_code,
                    'section_title': section['title'],
                    'page_number': section.get('page', 0)
                }
                concepts.append(concept)
        
        return concepts
    
    async def _build_concept_relationships(self, concepts: List[Dict]) -> List[Dict]:
        """建立概念间关系."""
        relationships = []
        
        # 基于共现分析建立关系
        for i, concept1 in enumerate(concepts):
            for concept2 in concepts[i+1:]:
                # 计算相似度
                similarity = self._calculate_similarity(concept1, concept2)
                
                # 领域自适应阈值，避免固定阈值导致跨课程误连
                threshold = self._relation_threshold_for_domain(concept1.get('course_code', 'general'))
                if similarity > threshold:
                    rel_type = "RELATED_TO" if similarity < 0.8 else "PREREQUISITE_OF"
                    relationships.append({
                        'from_id': concept1['id'],
                        'to_id': concept2['id'],
                        'type': rel_type,
                        'properties': {
                            'strength': similarity,
                            'created_at': datetime.now().isoformat()
                        }
                    })
        
        return relationships

    def _relation_threshold_for_domain(self, course_code: str) -> float:
        mapping = {
            'medical': 0.50,
            'education': 0.55,
            'general': 0.60,
            'casual': 0.70,
        }
        return mapping.get(course_code, mapping['general'])
    
    async def _create_concepts(self, session, concepts: List[Dict], course_code: str):
        """创建概念节点."""
        query = """
        UNWIND $concepts as concept
        MERGE (c:Concept {id: concept.id})
        SET c += concept
        SET c.created_at = timestamp()
        WITH c, concept
        MATCH (course:Course {code: concept.course_code})
        MERGE (c)-[:BELONGS_TO_COURSE]->(course)
        """
        
        await session.run(query, concepts=concepts)
```

```python
# app/services/graph/builders/code_builder.py
from tree_sitter import Language, Parser
import tree_sitter_python as tspython
from pathlib import Path

class CodeGraphBuilder:
    """代码知识图谱构建器."""
    
    def __init__(self):
        self.python_parser = self._create_parser(tspython.language())
    
    def _create_parser(self, language: Language) -> Parser:
        """创建解析器."""
        parser = Parser()
        parser.set_language(language)
        return parser
    
    async def build_from_directory(
        self,
        dir_path: str,
        project_id: str,
        language: str = 'python'
    ) -> Dict:
        """从代码目录构建图谱."""
        
        code_files = list(Path(dir_path).rglob(f"*.{language}"))
        
        all_code_units = []
        all_dependencies = []
        
        for file_path in code_files:
            # 解析单个文件
            result = await self._parse_file(str(file_path), project_id)
            all_code_units.extend(result['code_units'])
            all_dependencies.extend(result['dependencies'])
        
        # 批量写入图谱
        async for session in get_neo4j_session():
            await self._create_code_units(session, all_code_units)
            await self._create_dependencies(session, all_dependencies)
        
        return {
            'files_processed': len(code_files),
            'code_units_created': len(all_code_units),
            'dependencies_created': len(all_dependencies)
        }
    
    def _parse_file(self, file_path: str, project_id: str) -> Dict:
        """解析单个代码文件."""
        with open(file_path, 'rb') as f:
            source_code = f.read()
        
        tree = self.python_parser.parse(source_code)
        root_node = tree.root_node
        
        code_units = []
        dependencies = []
        
        # 遍历 AST
        self._traverse_ast(
            node=root_node,
            source_code=source_code,
            file_path=file_path,
            project_id=project_id,
            code_units=code_units,
            dependencies=dependencies
        )
        
        return {
            'code_units': code_units,
            'dependencies': dependencies
        }
    
    def _traverse_ast(
        self,
        node,
        source_code: bytes,
        file_path: Path,
        project_id: str,
        code_units: list,
        dependencies: list
    ):
        """遍历 AST提取代码单元."""
        
        if node.type == 'function_definition':
            code_unit = self._extract_function(node, source_code, file_path, project_id)
            code_units.append(code_unit)
        
        elif node.type == 'class_definition':
            code_unit = self._extract_class(node, source_code, file_path, project_id)
            code_units.append(code_unit)
        
        # 递归遍历子节点
        for child in node.children:
            self._traverse_ast(
                child, source_code, file_path, project_id,
                code_units, dependencies
            )
    
    def _extract_function(
        self,
        node,
        source_code: bytes,
        file_path: Path,
        project_id: str
    ) -> Dict:
        """提取函数定义."""
        name_node = node.child_by_field_name('name')
        params_node = node.child_by_field_name('parameters')
        
        return {
            'id': f"code_{project_id}_{file_path.stem}_{name_node.text.decode()}",
            'type': 'function',
            'name': name_node.text.decode(),
            'file_path': str(file_path),
            'start_line': node.start_point[0],
            'end_line': node.end_point[0],
            'signature': params_node.text.decode() if params_node else '',
            'body': node.child_by_field_name('body').text.decode(),
            'language': 'python',
            'complexity_cyclomatic': self._calculate_complexity(node),
            'loc': node.end_point[0] - node.start_point[0]
        }
```

```python
# app/services/graph/maintenance.py
class GraphMaintenance:
    """图谱维护服务."""
    
    async def incremental_update(
        self,
        file_path: str,
        old_content: str,
        new_content: str
    ) -> Dict:
        """增量更新图谱."""
        
        # 1. 解析新旧代码
        old_units = await self._parse_code(old_content, file_path)
        new_units = await self._parse_code(new_content, file_path)
        
        # 2. 计算差异
        added_units = self._find_added(old_units, new_units)
        removed_units = self._find_removed(old_units, new_units)
        modified_units = self._find_modified(old_units, new_units)
        
        # 3. 执行更新
        async for session in get_neo4j_session():
            # 删除已移除的
            await self._delete_code_units(session, removed_units)
            
            # 更新已修改的
            await self._update_code_units(session, modified_units)
            
            # 创建新增的
            await self._create_code_units(session, added_units)
        
        return {
            'added': len(added_units),
            'removed': len(removed_units),
            'modified': len(modified_units)
        }
    
    async def cleanup_orphan_nodes(self) -> int:
        """清理孤立节点."""
        query = """
        MATCH (n)
        WHERE NOT (n)--()
        AND n.label IN ['Concept', 'CodeUnit', 'BugCase']
        DETACH DELETE n
        """
        
        async for session in get_neo4j_session():
            result = await session.run(query)
            return result.consume().counters.nodes_deleted()
        
        return 0
    
    async def rebuild_indexes(self):
        """重建索引."""
        queries = [
            "CREATE INDEX concept_id_index IF NOT EXISTS FOR (c:Concept) ON (c.id)",
            "CREATE INDEX code_file_path_index IF NOT EXISTS FOR (code:CodeUnit) ON (code.file_path)",
            "DROP INDEX concept_name_index IF EXISTS",
            "CREATE FULLTEXT INDEX concept_name_index FOR (c:Concept) ON EACH [c.name, c.description]"
        ]
        
        async for session in get_neo4j_session():
            for query in queries:
                await session.run(query)
```

```python
class GraphQualityChecker:
    """图谱质量检查器."""
    
    async def check_quality(self) -> Dict:
        """执行质量检查."""
        
        checks = {
            'orphan_nodes': await self._check_orphan_nodes(),
            'circular_dependencies': await self._check_circular_dependencies(),
            'missing_properties': await self._check_missing_properties(),
            'duplicate_concepts': await self._check_duplicate_concepts(),
            'memory_link_consistency': await self._check_memory_link_consistency()
        }
        
        return {
            'score': self._calculate_score(checks),
            'issues': checks,
            'recommendations': self._generate_recommendations(checks)
        }
    
    async def _check_orphan_nodes(self) -> List[Dict]:
        """检查孤立节点."""
        query = """
        MATCH (n)
        WHERE NOT (n)--()
        AND n.label IN ['Concept', 'CodeUnit', 'BugCase']
        RETURN n.id as id, n.label as label, n.name as name
        """
        
        async for session in get_neo4j_session():
            result = await session.run(query)
            return [record.data() async for record in result]
        
        return []
    
    async def _check_duplicate_concepts(self) -> List[Dict]:
        """检查重复概念."""
        query = """
        MATCH (c:Concept)
        WITH c.name as name, collect(c) as concepts
        WHERE size(concepts) > 1
        UNWIND concepts as c
        RETURN c.id as id, c.name as name, size(concepts) as duplicates
        """
        
        async for session in get_neo4j_session():
            result = await session.run(query)
            return [record.data() async for record in result]
        
        return []

    async def _check_memory_link_consistency(self) -> List[Dict]:
        """检查记忆引用链是否存在断裂."""
        query = """
        MATCH (m:Memory)-[:MENTIONS]->(c:Concept)
        WHERE c.id IS NULL OR c.name IS NULL
        RETURN m.id as memory_id, c.id as concept_id
        LIMIT 100
        """

        async for session in get_neo4j_session():
            result = await session.run(query)
            return [record.data() async for record in result]

        return []
```

### 5.2 缓存策略

```python
# app/utils/graph_cache.py
import redis
import json
from typing import Optional, Any
from functools import wraps

class GraphCache:
    """图谱查询缓存."""
    
    def __init__(self, redis_url: str):
        self.redis = redis.from_url(redis_url)
        self.default_ttl = 3600  # 1小时
    
    def cache_query(self, key_prefix: str, ttl: Optional[int] = None):
        """查询缓存装饰器."""
        def decorator(func):
            @wraps(func)
            async def wrapper(*args, **kwargs):
                # 生成缓存键
                cache_key = f"{key_prefix}:{func.__name__}:{str(kwargs)}"
                
                # 尝试从缓存读取
                cached = self.redis.get(cache_key)
                if cached:
                    return json.loads(cached)
                
                # 执行查询
                result = await func(*args, **kwargs)
                
                # 写入缓存
                self.redis.setex(
                    cache_key,
                    ttl or self.default_ttl,
                    json.dumps(result, default=str)
                )
                
                return result
            return wrapper
        return decorator

# 使用示例
graph_cache = GraphCache(settings.REDIS_URL)

@graph_cache.cache_query("concept", ttl=7200)
async def get_concept_by_id(concept_id: str):
    query = GraphQueries.FIND_CONCEPT_BY_ID
    async for session in get_neo4j_session():
        result = await session.run(query, concept_id=concept_id)
        return await result.single()
```
