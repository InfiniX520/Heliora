```python
# scripts/export_config.py
#!/usr/bin/env python3
"""导出配置到加密文件."""

import os
import json
import getpass
from cryptography.fernet import Fernet
from pathlib import Path

def generate_key() -> str:
    """生成加密密钥."""
    return Fernet.generate_key().decode()

def export_config(output_file: str = "config.enc"):
    """导出加密配置."""
    
    # 需要导出的环境变量
    sensitive_vars = [
        'DATABASE_URL',
        'NEO4J_URI',
        'NEO4J_USER',
        'NEO4J_PASSWORD',
        'REDIS_URL',
        'LLM_API_KEY',
        'LLM_FALLBACK_API_KEY',
        'JWT_SECRET_KEY',
        'SENTRY_DSN'
    ]
    
    # 收集配置
    config = {}
    for var in sensitive_vars:
        value = os.environ.get(var)
        if value:
            config[var] = value
    
    # 生成密钥
    key = generate_key()
    fernet = Fernet(key.encode())
    
    # 加密配置
    config_json = json.dumps(config, indent=2)
    encrypted = fernet.encrypt(config_json.encode())
    
    # 保存加密文件
    with open(output_file, 'wb') as f:
        f.write(encrypted)
    
    # 保存密钥（单独保存）
    key_file = output_file.replace('.enc', '.key')
    with open(key_file, 'w') as f:
        f.write(key)
    
    print(f" 配置已导出到：{output_file}")
    print(f" 密钥已保存到：{key_file}")
    print("\n  重要提示:")
    print("1. 请将密钥文件单独保存（建议使用密码管理器）")
    print("2. 不要将密钥文件提交到版本控制")
    print("3. 在新机器上使用import_config.py 导入配置")

if __name__ == "__main__":
    export_config()
```

### 2.2 配置导入脚本

```python
# scripts/import_config.py
#!/usr/bin/env python3
"""从加密文件导入配置."""

import os
import json
from cryptography.fernet import Fernet
from pathlib import Path

def import_config(encrypted_file: str = "config.enc", key_file: str = "config.key"):
    """导入加密配置."""
    
    # 读取密钥
    key_path = Path(key_file)
    if not key_path.exists():
        print(f" 密钥文件不存在：{key_file}")
        return
    
    with open(key_path, 'r') as f:
        key = f.read().strip()
    
    # 读取加密配置
    encrypted_path = Path(encrypted_file)
    if not encrypted_path.exists():
        print(f" 加密配置文件不存在：{encrypted_file}")
        return
    
    with open(encrypted_path, 'rb') as f:
        encrypted = f.read()
    
    # 解密配置
    fernet = Fernet(key.encode())
    decrypted = fernet.decrypt(encrypted)
    config = json.loads(decrypted.decode())
    
    # 写入.env 文件
    env_file = Path(".env")
    env_content = ""
    
    if env_file.exists():
        with open(env_file, 'r') as f:
            env_content = f.read()
        env_content += "\n# Imported config\n"
    
    for key, value in config.items():
        env_content += f"{key}={value}\n"
    
    with open(env_file, 'w') as f:
        f.write(env_content)
    
    print(f" 配置已导入到：{env_file}")
    print(f" 导入了{len(config)} 个配置项")
    print("\n  请检查.env文件并修改机器特定的配置（如localhost 地址）")

if __name__ == "__main__":
    import_config()
```

```python
# scripts/check_env.py
#!/usr/bin/env python3
"""检查开发环境配置."""

import sys
import subprocess
from pathlib import Path

def check_python():
    """检查Python版本."""
    try:
        import sys
        version = sys.version_info
        if version.major == 3 and version.minor >= 11:
            print(f" Python {version.major}.{version.minor}.{version.micro}")
            return True
        else:
            print(f" Python 版本过低(需要 3.11+, 当前 {version.major}.{version.minor})")
            return False
    except Exception as e:
        print(f" Python检查失败：{e}")
        return False

def check_node():
    """检查Node.js 版本."""
    try:
        result = subprocess.run(['node', '--version'], capture_output=True, text=True)
        version = result.stdout.strip()
        print(f" Node.js {version}")
        return True
    except Exception as e:
        print(f" Node.js 未安装：{e}")
        return False

def check_docker():
    """检查Docker."""
    try:
        result = subprocess.run(['docker', '--version'], capture_output=True, text=True)
        version = result.stdout.strip()
        print(f" Docker {version}")
        return True
    except Exception as e:
        print(f" Docker未安装：{e}")
        return False

def check_containers():
    """检查Docker 容器."""
    containers = ['neo4j', 'redis', 'heliora-db']
    for container in containers:
        try:
            result = subprocess.run(
                ['docker', 'ps', '-f', f'name={container}', '--format', '{{.Names}}'],
                capture_output=True, text=True
            )
            if container in result.stdout:
                print(f" 容器{container} 运行中")
            else:
                print(f"  容器{container} 未运行")
        except Exception as e:
            print(f" 容器 {container} 检查失败：{e}")

def check_env_file():
    """检查.env文件."""
    env_file = Path('.env')
    if env_file.exists():
        print(f" .env 文件存在")
        
        # 检查必要变量
        required_vars = [
            'DATABASE_URL',
            'NEO4J_URI',
            'NEO4J_USER',
            'NEO4J_PASSWORD',
            'LLM_API_KEY',
            'JWT_SECRET_KEY'
        ]
        
        with open(env_file, 'r') as f:
            content = f.read()
        
        for var in required_vars:
            if var in content:
                print(f"    {var} 已配置")
            else:
                print(f"    {var} 未配置")
    else:
        print(f" .env 文件不存在")

def main():
    """主函数."""
    print("=" * 50)
    print("  Heliora 环境检查")
    print("=" * 50)
    print()
    
    checks = [
        ("Python", check_python),
        ("Node.js", check_node),
        ("Docker", check_docker),
    ]
    
    results = []
    for name, check in checks:
        print(f"检查 {name}...")
        results.append(check())
        print()
    
    check_containers()
    print()
    check_env_file()
    print()
    
    if all(results):
        print(" 所有检查通过！")
        return 0
    else:
        print("  部分检查未通过，请修复后继续")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```

```python
# scripts/generate_secrets.py
#!/usr/bin/env python3
"""生成安全的随机密钥."""

import secrets
import string

def generate_jwt_secret() -> str:
    """生成JWT密钥."""
    return secrets.token_urlsafe(32)

def generate_password(length: int = 32) -> str:
    """生成随机密码."""
    alphabet = string.ascii_letters + string.digits + "!@#$%^&*"
    return ''.join(secrets.choice(alphabet) for _ in range(length))

def main():
    """生成所有密钥."""
    print("=" * 50)
    print("  Heliora 密钥生成器")
    print("=" * 50)
    print()
    
    secrets_dict = {
        'JWT_SECRET_KEY': generate_jwt_secret(),
        'DATABASE_PASSWORD': generate_password(),
        'NEO4J_PASSWORD': generate_password(),
        'REDIS_PASSWORD': generate_password()
    }
    
    for name, value in secrets_dict.items():
        print(f"{name}={value}")
    
    print()
    print("  请将以上密钥保存到安全位置（如密码管理器）")
```
