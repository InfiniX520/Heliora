## 1. 核心安全原则

### 1.1 设计哲学

```
用户授权优先 ❚❚ 本地执行 ❚❚ 可回退 ❚❚ 透明可查
```

**四大原则**:

1. **用户授权优先**: 任何文件操作前必须获得用户明确授权
2. **本地执行**: 所有文件操作在本地完成，不上传云端
3. **可回退**: 任何操作都支持撤销和恢复
4. **透明可查**: 所有操作记录日志，用户可随时查看

### 1.2 安全模式开关

1. `strict`（默认）:
    - 执行完整安全策略（系统目录拦截、高风险二次确认、参数白名单）。
2. `trusted_local_max`（可选）:
    - 面向本地部署高权限使用场景，允许放宽部分拦截规则。
    - 启用条件：`SECURITY_POLICY_MODE=trusted_local_max` 且 `LOCAL_MAX_PRIVILEGE_ACK=true`。
    - 强制要求：仍保留操作确认、备份、回滚与审计链路。

```python
# 文件夹扫描授权管理器
class ScanPermissionManager:
    """扫描权限管理器."""

    def __init__(self, security_mode: str = "strict"):
        self.security_mode = security_mode
    
    # 禁止扫描的系统目录
    FORBIDDEN_PATHS = [
        r'C:\Windows',
        r'C:\Program Files',
        r'C:\Program Files (x86)',
        r'C:\Users\*\AppData',
        r'/System',
        r'/Applications',
        r'/Library',
        r'/usr',
        r'/bin',
        r'/sbin',
    ]
    
    # 允许扫描的文件类型
    ALLOWED_EXTENSIONS = [
        # 文档
        '.pdf', '.doc', '.docx', '.ppt', '.pptx', '.xls', '.xlsx',
        '.txt', '.md', '.rtf',
        
        # 代码
        '.py', '.java', '.cpp', '.c', '.h', '.js', '.ts', '.go',
        '.rs', '.rb', '.php', '.swift', '.kt',
        
        # 图片(支持OCR)
        '.png', '.jpg', '.jpeg', '.gif', '.bmp',
        
        # 其他
        '.json', '.xml', '.yaml', '.yml', '.csv',
    ]
    
    async def request_permission(self, folder_path: str) -> bool:
        """请求扫描许可."""
        
        # 1. strict 模式下检查系统目录；trusted_local_max 允许放宽该限制
        if self.security_mode == "strict" and self._is_forbidden_path(folder_path):
            raise SecurityError(
                f"不能扫描系统目录：{folder_path}"
            )
        
        # 2. 显示扫描说明
        message = f"""
         检测到文件夹：{folder_path}
        
        我可以帮您扫描里面的文件，并建议整理方式。
        
        扫描说明:
        ✓ 仅在本地进行，不会上传任何内容
        ✓ 读取文件内容以提取关键词和标题
        ✓ 不会修改或删除原文件
        ✓ 扫描完成后您可以选择是否整理
        
        允许我扫描吗？
        """
        
        # 3. 等待用户确认
        return await user_confirm(message)
    
    def _is_forbidden_path(self, path: str) -> bool:
        """检查是否是禁止的路径."""
        import fnmatch
        
        for forbidden in self.FORBIDDEN_PATHS:
            if fnmatch.fnmatch(path, forbidden):
                return True
        return False
```

### 3.2 重命名执行器

```python
# 安全的文件重命名执行器
class SafeFileRenamer:
    """安全文件重命名执行器."""
    
    def __init__(self, backup_dir: str):
        self.backup_dir = backup_dir
        self.backup_map = {}  # 原文件路径 → 备份路径
        self.operation_log = []  # 操作日志
    
    async def execute(self, rename_plan: List[RenamePlan]) -> bool:
        """执行重命名计划."""
        
        try:
            # Step 1: 创建所有文件的备份
            logger.info("开始创建备份...")
            for plan in rename_plan:
                backup_path = await self._create_backup(plan.original_path)
                self.backup_map[plan.original_path] = backup_path
            
            logger.info(f"已为{len(self.backup_map)} 个文件创建备份")
            
            # Step 2: 逐个执行重命名
            success_count = 0
            for plan in rename_plan:
                success = await self._rename_single_file(plan)
                if success:
                    success_count += 1
                    self.operation_log.append({
                        'action': 'rename',
                        'from': plan.original_path,
                        'to': plan.target_path,
                        'backup': self.backup_map[plan.original_path],
                        'status': 'success',
                    })
                else:
                    self.operation_log.append({
                        'action': 'rename',
                        'from': plan.original_path,
                        'to': plan.target_path,
                        'status': 'failed',
                    })
            
            logger.info(f"重命名完成：{success_count}/{len(rename_plan)}")
            
            # Step 3: 生成映射表
            await self._generate_mapping_table()
            
            return True
            
        except Exception as e:
            # 发生错误时回滚
            logger.error(f"重命名失败：{e}")
            await self.rollback()
            raise
    
    async def _create_backup(self, file_path: str) -> str:
        """创建文件备份."""
        import shutil
        from pathlib import Path
        
        # 确保备份目录存在
        Path(self.backup_dir).mkdir(parents=True, exist_ok=True)
        
        # 生成备份文件名 (带时间戳)
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_filename = f"{Path(file_path).stem}_{timestamp}{Path(file_path).suffix}"
        backup_path = Path(self.backup_dir) / backup_filename
        
        # 复制文件
        shutil.copy2(file_path, backup_path)
        
        return str(backup_path)
    
    async def _rename_single_file(self, plan: RenamePlan) -> bool:
        """重命名单个文件."""
        import shutil
        
        try:
            # 1. 复制并重命名
            shutil.copy2(plan.original_path, plan.target_path)
            
            # 2. 验证新文件
            if not os.path.exists(plan.target_path):
                raise Exception("新文件创建失败")
            
            # 3. 删除原文件
            os.remove(plan.original_path)
            
            return True
            
        except Exception as e:
            logger.error(f"重命名失败{plan.original_path}: {e}")
            return False
    
    async def rollback(self):
        """回滚所有操作."""
        logger.info("开始回滚...")
        
        for original_path, backup_path in self.backup_map.items():
            if os.path.exists(backup_path):
                shutil.copy2(backup_path, original_path)
                logger.info(f"已恢复：{original_path}")
        
        logger.info("回滚完成")
```

### 3.3 冲突处理

```python
# 文件冲突处理器
class ConflictResolver:
    """处理文件命名冲突."""
    
    async def handle_conflict(self, target_path: str) -> ConflictAction:
        """处理冲突."""
        
        # 检测目标文件是否存在
        if not os.path.exists(target_path):
            return ConflictAction.PROCEED  # 无冲突，继续
        
        # 计算文件哈希，检查是否完全相同
        existing_hash = await calculate_hash(target_path)
        new_hash = await calculate_hash(self.new_file_path)
        
        if existing_hash == new_hash:
            # 文件内容完全相同
            return ConflictAction.SKIP  # 跳过，无需处理
        
        # 文件不同，询问用户
        message = f"""
          目标文件已存在：{target_path}
        
        选项:
        [1]覆盖 - 用新文件替换旧文件(旧文件会先备份)
        [2] 保留两者 - 新文件重命名为"文件名_1.pdf"
        [3]跳过 - 保持原文件不变
        [4]取消整个操作
        
        两个文件内容不同，请谨慎选择。
        """
        
        return await user_select(message)
```

```python
# 备份恢复管理器
class BackupRestorer:
    """备份恢复管理器."""
    
    def __init__(self, backup_dir: str):
        self.backup_dir = backup_dir
    
    async def list_restoreable(self) -> List[BackupItem]:
        """列出可恢复的文件."""
        import glob
        
        # 扫描备份目录
        backup_files = glob.glob(
            os.path.join(self.backup_dir, "*_*_*.pdf")
        )
        
        # 解析备份信息
        restoreable = []
        for backup_path in backup_files:
            info = await self._parse_backup_info(backup_path)
            restoreable.append(info)
        
        return restoreable
    
    async def restore(self, backup_items: List[str]) -> int:
        """恢复选中的文件."""
        restored_count = 0
        
        for backup_path in backup_items:
            info = await self._parse_backup_info(backup_path)
            
            # 恢复文件
            shutil.copy2(backup_path, info.original_path)
            restored_count += 1
        
        return restored_count
    
    async def _parse_backup_info(self, backup_path: str) -> BackupItem:
        """解析备份文件信息."""
        from pathlib import Path
        
        filename = Path(backup_path).stem
        # 格式：原文件名_时间戳
        parts = filename.rsplit('_', 2)
        
        if len(parts) >= 3:
            original_name = '_'.join(parts[:-2])
            timestamp_str = '_'.join(parts[-2:])
            timestamp = datetime.strptime(
                timestamp_str, "%Y%m%d_%H%M%S"
            )
        else:
            original_name = filename
            timestamp = None
        
        return BackupItem(
            backup_path=backup_path,
            original_name=original_name,
            backup_time=timestamp,
            file_size=os.path.getsize(backup_path),
        )
```

```python
class StorageMonitor:
    """存储空间监控器."""
    
    def __init__(self, base_path: str, threshold_gb: float = 5.0):
        self.base_path = base_path
        self.threshold = threshold_gb * 1024 * 1024 * 1024
    
    async def check_and_alert(self) -> Optional[AlertMessage]:
        """检查空间并生成警报."""
        
        # 计算已用空间
        used_space = get_folder_size(self.base_path)
        
        # 计算磁盘剩余空间
        free_space = get_disk_free_space(self.base_path)

        if used_space > self.threshold:
            return AlertMessage(
                type='threshold_exceeded',
                title='存储空间快满啦',
                content=f"""
                你的Heliora知识库已占用{used_space/1024/1024/1024:.1f}GB
                (阈值：{self.threshold/1024/1024/1024:.1f}GB)
                
                我帮你看了下，这些文件可能可以清理：
                - 30 天未打开的文件：{old_files_count}个
                - 重复文件：{duplicate_count}个
                
                要我帮你整理一下吗？
                """,
                actions=['清理', '稍后', '忽略']
            )
        
        # 检查磁盘剩余空间
        if free_space < self.threshold * 0.2:  # 剩余<20%

                type='low_disk_space',
                title='磁盘空间不足',

                你的磁盘剩余空间不足
```

```python
# 操作日志记录器
class OperationLogger:
    """操作日志记录器."""
    
    def __init__(self, log_path: str):
        self.log_path = log_path
    
    def log(self, operation: Operation):
        """记录操作日志."""
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'action': operation.action,  # scan/rename/delete/restore
            'source': operation.source_path,
            'target': operation.target_path,
            'backup': operation.backup_path,
            'status': operation.status,  # success/failed/cancelled
            'user_confirmed': operation.user_confirmed,
            'details': operation.details,
        }
        
        # 写入日志文件
        with open(self.log_path, 'a', encoding='utf-8') as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + '\n')
```
